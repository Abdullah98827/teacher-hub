create view public.active_memberships_view as
select
  teachers.id as teacher_id,
  teachers.email,
  teachers.approved,
  memberships.tier,
  memberships.subject_ids,
  memberships.active,
  memberships.created_at,
  ARRAY(
    select
      subjects.name
    from
      subjects
    where
      subjects.id = any (memberships.subject_ids)
  ) as subject_names
from
  memberships
  join teachers on memberships.id = teachers.id
where
  memberships.active = true;

create view public.user_roles_details as
select
  ur.id,
  ur.role,
  COALESCE(t.email, a.email) as email,
  COALESCE(t.first_name, a.first_name) as first_name,
  COALESCE(t.last_name, a.last_name) as last_name,
  ur.created_at,
  ur.updated_at
from
  user_roles ur
  left join teachers t on ur.id = t.id
  left join admins a on ur.id = a.id;
