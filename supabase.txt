create view public.active_memberships_view as
select
  teachers.id as teacher_id,
  teachers.email,
  teachers.approved,
  memberships.tier,
  memberships.subject_ids,
  memberships.active,
  memberships.created_at,
  ARRAY(
    select
      subjects.name
    from
      subjects
    where
      subjects.id = any (memberships.subject_ids)
  ) as subject_names
from
  memberships
  join teachers on memberships.id = teachers.id
where
  memberships.active = true;

create view public.resource_share_stats as
select
  resource_id,
  count(*) as total_shares,
  count(distinct shared_by) as unique_sharers,
  count(distinct shared_with) as unique_recipients,
  count(
    case
      when share_method = 'direct_message'::text then 1
      else null::integer
    end
  ) as dm_shares,
  count(
    case
      when share_method = 'link_copy'::text then 1
      else null::integer
    end
  ) as link_shares,
  max(created_at) as last_shared
from
  resource_shares
group by
  resource_id;

  constraint user_roles_id_fkey foreign KEY (id) references auth.users (id) on delete CASCADE,
  constraint user_roles_role_check check (
    (
      role = any (
        array[
          'teacher'::text,
          'admin'::text,
          'super_admin'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;










create view public.user_roles_details as
select
  ur.id,
  ur.role,
  COALESCE(t.email, a.email) as email,
  COALESCE(t.first_name, a.first_name) as first_name,
  COALESCE(t.last_name, a.last_name) as last_name,
  ur.created_at,
  ur.updated_at
from
  user_roles ur
  left join teachers t on ur.id = t.id
  left join admins a on ur.id = a.id;






  create table public.users (
  id uuid not null,
  email text null,
  name text null,
  created_at timestamp without time zone null default now(),
  constraint users_pkey primary key (id),
  constraint users_id_fkey foreign KEY (id) references auth.users (id)
) TABLESPACE pg_default;





create table user_reports (
  id uuid primary key default gen_random_uuid(),

  reported_user_id uuid not null references teachers(id),
  reason text not null,
  details text,
  status text default 'pending' check (status in ('pending', 'reviewed', 'resolved', 'dismissed')),
  admin_notes text,
  created_at timestamp default now(),
  updated_at timestamp default now()
